#include <stdint.h>

/* PTE bit definitions */
#define PTE_PRESENT   (1ULL << 0)   /* page is in RAM              */
#define PTE_WRITABLE  (1ULL << 1)   /* 0 = read-only               */
#define PTE_USER      (1ULL << 2)   /* 1 = user-accessible         */
#define PTE_NX        (1ULL << 63)  /* execute disable             */
#define PTE_ADDR_MASK 0x000FFFFFFFFFF000ULL
#define HHDM_OFFSET   0xFFFF800000000000ULL /* kernel's physical map base  */

typedef uint64_t pte_t;

/* Convert a physical address to a kernel-accessible virtual address */
static inline pte_t* phys_to_virt(uint64_t paddr) {
    return (pte_t*)(paddr + HHDM_OFFSET);
}

/* Walk page tables and return a pointer to the PTE for vaddr.
 * cr3_phys: value read from CR3 (physical address of PML4).
 * Returns NULL if any level is not present. */
pte_t* walk_to_pte(uint64_t cr3_phys, uint64_t vaddr) {
    pte_t *pml4 = phys_to_virt(cr3_phys & PTE_ADDR_MASK);
    pte_t  e4   = pml4[(vaddr >> 39) & 0x1FF];
    if (!(e4 & PTE_PRESENT)) return 0;

    pte_t *pdpt = phys_to_virt(e4 & PTE_ADDR_MASK);
    pte_t  e3   = pdpt[(vaddr >> 30) & 0x1FF];
    if (!(e3 & PTE_PRESENT)) return 0;

    pte_t *pd   = phys_to_virt(e3 & PTE_ADDR_MASK);
    pte_t  e2   = pd[(vaddr >> 21) & 0x1FF];
    if (!(e2 & PTE_PRESENT)) return 0;

    pte_t *pt   = phys_to_virt(e2 & PTE_ADDR_MASK);
    return &pt[(vaddr >> 12) & 0x1FF];
}

/* Mark a page as read-only and flush its TLB entry */
void make_page_readonly(uint64_t cr3_phys, uint64_t vaddr) {
    pte_t *pte = walk_to_pte(cr3_phys, vaddr);
    if (!pte) return;

    *pte &= ~PTE_WRITABLE;                       /* clear R/W bit */
    /* memory fence + TLB flush for this specific page */
    __asm__ volatile(
        "mfence\n\t"
        "invlpg (%0)"
        :: "r"(vaddr) : "memory"
    );
}

/* Mark a page as non-executable (set NX bit) */
void make_page_noexec(uint64_t cr3_phys, uint64_t vaddr) {
    pte_t *pte = walk_to_pte(cr3_phys, vaddr);
    if (!pte) return;

    *pte |= PTE_NX;                               /* set execute-disable bit */
    __asm__ volatile("mfence\n\tinvlpg (%0)" :: "r"(vaddr) : "memory");
}

Por que o mfence antes do invlpg?

Processadores x86 possuem um store buffer вҖ” escritas em memГіria podem nГЈo ser visГӯveis imediatamente para outros componentes do pipeline. Sem a barreira de memГіria (mfence), a MMU poderia executar invlpg antes que a escrita na PTE tivesse propagado para o cache, resultando em uma TLB com a nova traduГ§ГЈo mas uma PTE ainda com o valor antigo. O mfence garante que o store seja drenado antes da invalidaГ§ГЈo.

x86 processors have a store buffer вҖ” memory writes may not be immediately visible to other pipeline components. Without the memory barrier (mfence), the MMU could execute invlpg before the PTE write had propagated to cache, leaving the TLB with the new translation but the PTE still holding the old value. The mfence ensures the store is drained before the invalidation.
