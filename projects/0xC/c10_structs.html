O que são Structs?

Uma struct é um tipo composto que agrupa variáveis de tipos diferentes sob um único nome. Enquanto um array armazena elementos do mesmo tipo, uma struct pode combinar int, char, double e ponteiros em um único bloco de memória.

cstruct_basico.c
#include <stdio.h>
#include <string.h>

/* Definição do tipo — não aloca memória */
typedef struct {
    unsigned int  id;
    char          nome[32];
    double        saldo;
    int           ativo;
} Conta;

int main(void) {
    Conta c1;                          /* declara na stack */

    c1.id    = 1001;
    strncpy(c1.nome, "Robert Lages", sizeof(c1.nome) - 1);
    c1.nome[sizeof(c1.nome) - 1] = '\0'; /* garante terminação */
    c1.saldo  = 1500.75;
    c1.ativo  = 1;

    printf("ID: %u | Nome: %s | Saldo: R$%.2f\n",
           c1.id, c1.nome, c1.saldo);
    printf("sizeof(Conta) = %zu bytes\n", sizeof(Conta));

    return 0;
}

Memory Padding e Alinhamento

A CPU x86-64 acessa a memória de forma mais eficiente quando os dados estão em endereços alinhados ao seu tamanho — um double (8 bytes) deve estar em um endereço múltiplo de 8, um int (4 bytes) em múltiplo de 4. Para garantir isso, o compilador insere bytes de preenchimento (padding) entre os membros da struct.

cpadding_demo.c
/* Layout com padding ruim (36 bytes) */
struct Ruim {
    char   a;    /* 1 byte  + 7 bytes padding */
    double b;    /* 8 bytes */
    char   c;    /* 1 byte  + 7 bytes padding */
    double d;    /* 8 bytes */
}; /* sizeof = 32 bytes (com padding) */

/* Layout otimizado — maiores primeiro (24 bytes) */
struct Bom {
    double b;    /* 8 bytes */
    double d;    /* 8 bytes */
    char   a;    /* 1 byte */
    char   c;    /* 1 byte  + 6 bytes padding final */
}; /* sizeof = 24 bytes (com padding) */
struct Ruim:
offset 0a (1B)
offset 1-7PADDING← 7 bytes desperdiçados para alinhar b
offset 8b (8B)
offset 16c (1B)
offset 17-23PADDING← 7 bytes desperdiçados para alinhar d
offset 24d (8B)
Total: 32B← 14 bytes de padding!

Verificando o layout real com offsetof

coffsetof_demo.c
#include <stddef.h>  /* offsetof */
#include <stdio.h>

typedef struct {
    char   flag;
    int    valor;
    double preco;
} Produto;

int main(void) {
    printf("sizeof  : %zu\n", sizeof(Produto));
    printf("flag  @ +%zu\n", offsetof(Produto, flag));
    printf("valor @ +%zu\n", offsetof(Produto, valor));
    printf("preco @ +%zu\n", offsetof(Produto, preco));
    return 0;
}
/* Output:
   sizeof  : 16
   flag  @ +0
   valor @ +4   ← 3 bytes de padding entre flag e valor
   preco @ +8                                            */

Passagem por referência com structs

Passar uma struct por valor copia todos os bytes para o stack frame da função chamada. Para structs grandes, isso é custoso. Passe por ponteiro:

cstruct_ponteiro.c
void depositar(Conta *c, double valor) {
    if (c == NULL || valor <= 0) return;
    c->saldo += valor; /* c->campo == (*c).campo */
}

/* No heap: */
Conta *nova = (Conta*) malloc(sizeof(Conta));
if (!nova) { /* handle error */ }
nova->id    = 2001;
nova->saldo = 0.0;
depositar(nova, 500.0);
printf("Saldo: %.2f\n", nova->saldo); /* 500.00 */
free(nova); nova = NULL;

// O operador ->: ptr->campo é syntactic sugar para (*ptr).campo. Ele desreferencia o ponteiro e acessa o campo em uma operação. Sempre use -> com ponteiros para struct — é mais legível e o compilador gera o mesmo código.

__attribute__((packed)) — removendo o padding

Em protocolos de rede, drivers e parsing de formatos binários, você precisa que a struct espelhe exatamente o layout do wire/disco, sem padding. O atributo packed elimina o padding, mas acessa desalinhados podem ser mais lentos ou gerar faults em arquiteturas estritas (ARM, por exemplo).

cpacked_struct.c
typedef struct __attribute__((packed)) {
    uint8_t  tipo;    /* 1 byte */
    uint16_t tamanho; /* 2 bytes — sem padding! */
    uint32_t payload; /* 4 bytes */
} Pacote; /* sizeof = 7, não 8 */
